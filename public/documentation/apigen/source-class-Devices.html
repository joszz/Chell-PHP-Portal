 61     /**
 62      * Pings a device and returns the response time.
 63      *
 64      * @param mixed $host   Which host to ping
 65      * @param mixed $ttl    The TimeToLive for the ping request. Defaults to 1 second
 66      * @return bool|double  The time it took for the device to respond or false if failed.
 67      */
 68     private function pingExec($host, $ttl = 10)
 69     {
 70         $latency = false;
 71         $ttl     = escapeshellcmd($ttl);
 72         $host    = escapeshellcmd($host);
 73 
 74         // Exec string for Windows-based systems.
 75         if (strtoupper(substr(PHP_OS, 0, 3)) === 'WIN')
 76         {
 77             // -n = number of pings; -i = ttl.
 78             $exec_string = 'ping -n 1 -w 1 -i ' . $ttl . ' ' . $host;
 79         }
 80         // Exec string for UNIX-based systems (Mac, Linux).
 81         else
 82         {
 83             // -n = numeric output; -c = number of pings; -t = ttl.
 84             $exec_string = 'ping -n -c 1 -w 1 -t ' . $ttl . ' ' . $host;
 85         }
 86 
 87         exec($exec_string, $output, $return);
 88 
 89         // Strip empty lines and reorder the indexes from 0 (to make results more
 90         // uniform across OS versions).
 91         $output = array_values(array_filter($output));
 92         // If the result line in the output is not empty, parse it.
 93         if (!empty($output[1]))
 94         {
 95             // Search for a 'time' value in the result line.
 96             $response = preg_match("/time(?:=|<)(?<time>[\.0-9]+)(?:|\s)ms/", $output[1], $matches);
 97             // If there's a result and it's greater than 0, return the latency.
 98             if ($response > 0 && isset($matches['time']))
 99             {
100                 $latency = round($matches['time']);
101             }
102         }
103 
104         return $latency;
105     }

107     /**
108      * Wakes up a device by MAC address.
109      *
110      * @param mixed $mac            The device to wake.
111      * @param mixed $socket_number  The port to send the magic packet to.
112      * @param mixed $repetition     The amount of repition of the MAC in the magic packet. Defaults to 16.
113      * @return bool                 Whether or not socket_sendto with magic packet succeeded.
114      */
115     public function wakeOnLan($mac, $config, $socket_number = '7', $repetition = 16)
116     {
117         $addr_byte = explode(':', $mac);
118         $hw_addr = '';
119         for ($a=0; $a <6; $a++) $hw_addr .= chr(hexdec($addr_byte[$a]));
120         $msg = chr(255).chr(255).chr(255).chr(255).chr(255).chr(255);
121         for ($a = 1; $a <= $repetition; $a++) $msg .= $hw_addr;
122 
123         // send it to the broadcast address using UDP
124         // SQL_BROADCAST option isn't help!!
125         $s = socket_create(AF_INET, SOCK_DGRAM, SOL_UDP);
126         if ($s == false)
127         {
128             echo 'Error creating socket!\n';
129             echo 'Error code is "' . socket_last_error($s) . '" - ' . socket_strerror(socket_last_error($s));
130             return FALSE;
131         }
132         else
133         {
134             // setting a broadcast option to socket:
135             $opt_ret = socket_set_option($s, 1, 6, TRUE);
136             if($opt_ret <0)
137             {
138                 echo 'setsockopt() failed, error: ' . $opt_ret . '\n';
139                 return false;
140             }
141             if(socket_sendto($s, $msg, strlen($msg), 0, $config->network->broadcast, $socket_number))
142             {
143                 echo 'Magic Packet sent successfully!';
144                 socket_close($s);
145                 return true;
146             }
147             else
148             {
149                 echo 'Magic packet failed!';
150                 return false;
151             }
152         }
153     }
