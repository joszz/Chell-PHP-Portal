 37     /**
 38      * Add all fields to the form and set form specific attributes.
 39      *
 40      * @param Object $device    The device to add the fields for.
 41      */
 42     private function addFIelds($device)
 43     {
 44         $name = new Text('devices[' . $device->id . '][name]');
 45         $name->setLabel('Devicename');
 46         $name->setFilters(array('striptags', 'string'));
 47         $name->setAttributes(array('class' => 'form-control'));
 48         $name->setDefault($device->name);
 49         $name->addValidators(array(new PresenceOf(array())));
 50 
 51         $ip = new Text('devices[' . $device->id . '][ip]');
 52         $ip->setLabel('IP');
 53         $ip->setFilters(array('striptags', 'string'));
 54         $ip->setAttributes(array('class' => 'form-control'));
 55         $ip->setDefault($device->ip);
 56         $ip->addValidator(new Regex(array('pattern' => '/^(?:[0-9]{1,3}\.){3}[0-9]{1,3}$/')));
 57 
 58         $mac = new Text('devices[' . $device->id . '][mac]');
 59         $mac->setLabel('MAC');
 60         $mac->setFilters(array('striptags', 'string'));
 61         $mac->setAttributes(array('class' => 'form-control'));
 62         $mac->setDefault($device->mac);
 63         $mac->addValidator(new Regex(array('pattern' => '/^([0-9A-Fa-f]{2}[:]){5}([0-9A-Fa-f]{2})$/')));
 64 
 65         $webtemp = new Text('devices[' . $device->id . '][webtemp]');
 66         $webtemp->setLabel('Webtemp path');
 67         $webtemp->setFilters(array('striptags', 'string'));
 68         $webtemp->setAttributes(array('class' => 'form-control'));
 69         $webtemp->setDefault($device->webtemp);
 70 
 71         $shutdownMethod = new Select(
 72             'devices[' . $device->id . '][shutdown_method]',
 73             array('none' => 'None', 'rpc' => 'RPC'),
 74             array(
 75                 'useEmpty'      => false,
 76             )
 77         );
 78         $shutdownMethod->setDefault($device->shutdown_method);
 79 
 80         $showDasboard = new Check('devices[' . $device->id . '][show_on_dashboard]');
 81         $showDasboard->setLabel('Show on dashboard');
 82         $showDasboard->setFilters(array('striptags', 'int'));
 83         $showDasboard->setAttributes(array('class' => 'form-control'));
 84         $showDasboard->setDefault($device->show_on_dashboard);

 86         $id = new Hidden('devices[' . $device->id . '][id]');

 98     /**

100      *

102      * @param   object    $entity   The entity to validate.

105     public function IsValid($data = null, $entity = null)
106     {

108 
109         foreach($data['devices'] as $id => $device)
110         {
111             $newDevice = new Devices();
112             $newDevice->id = $id;
113
